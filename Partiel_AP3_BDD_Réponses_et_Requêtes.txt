Requête 1 :

SELECT 
    a.nom,
    a.prenom,
    e.nom AS epreuve,
    SUM(n.valeur * c.poids) AS note_finale
FROM athlete a
JOIN run r ON r.id_athlete = a.id_athlete
JOIN epreuve e ON e.id_epreuve = r.id_epreuve
JOIN note n ON n.id_run = r.id_run
JOIN critere c ON c.id_critere = n.id_critere
GROUP BY a.id_athlete, e.id_epreuve;

Requête 2 :

SELECT 
    CONCAT(a.prenom, ' ', a.nom) AS athlete
FROM athlete a
JOIN participation p ON p.id_athlete = a.id_athlete
GROUP BY a.id_athlete
HAVING COUNT(p.id_epreuve) >= 2;

Requête 3 :

SELECT 
    a.nom,
    a.prenom,
    p.nom AS pays
FROM athlete a
JOIN participation pa 
    ON pa.id_athlete = a.id_athlete
JOIN pays p 
    ON p.id_pays = a.id_pays
LEFT JOIN run r 
    ON r.id_athlete = a.id_athlete
    AND r.id_epreuve = pa.id_epreuve
WHERE r.id_run IS NULL;

Aucun athlète prévu pour la run n’a été absent : ils ont tous bien participé.

Requête 4 :

DELIMITER $$

CREATE PROCEDURE classement_epreuve(IN p_id_epreuve INT)
BEGIN
    SELECT 
        a.nom,
        a.prenom,
        SUM(n.valeur * c.poids) AS note_finale
    FROM athlete a
    JOIN run r ON r.id_athlete = a.id_athlete
    JOIN note n ON n.id_run = r.id_run
    JOIN critere c ON c.id_critere = n.id_critere
    WHERE r.id_epreuve = p_id_epreuve
    GROUP BY a.id_athlete
    ORDER BY note_finale DESC;
END $$

CALL classement_epreuve(1);
CALL classement_epreuve(2);


DELIMITER ;

CREATE TRIGGER verification_note

        SET MESSAGE_TEXT = 'La note est entre 0 et 100';
    END IF;
END $$
